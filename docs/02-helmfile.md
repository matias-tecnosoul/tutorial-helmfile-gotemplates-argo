üîç Estructura del helmfile.yaml
# 1. Ambientes (opcional)

  dev:
    values: [...]
  production:
    values: [...]

# 2. Repositorios

  - name: repo-name

# 3. Releases

    namespace: namespace
    chart: repo/chart
    version: x.y.z
    values:
      - path/to/values.yaml
    condition: enabled.flag  # Opcional
    needs: [other-release]    # Opcional
    labels:                   # Opcional
      key: value
